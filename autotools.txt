-- Update speed label
local function updateSpeedLabel()
    SpeedLabel.Text = "Speed: " .. string.format("%.2f", useInterval) .. "s"
end

-- Speed control buttons
SpeedDecreaseButton.MouseButton1Click:Connect(function()
    useInterval = math.max(0.01, useInterval - 0.1)
    useInterval = math.floor(useInterval * 100) / 100 -- Round to 2 decimals
    updateSpeedLabel()
end)

SpeedIncreaseButton.MouseButton1Click:Connect(function()
    useInterval = math.min(2.0, useInterval + 0.1)
    useInterval = math.floor(useInterval * 100) / 100 -- Round to 2 decimals
    updateSpeedLabel()
end)

-- Collapse functionality
local collapsed = false
CollapseButton.MouseButton1Click:Connect(function()
    collapsed = not collapsed
    ContentFrame.Visible = not collapsed
    CollapseButton.Text = collapsed and "+" or "-"
    MainFrame.Size = collapsed and UDim2.new(0, 180, 0, 22) or UDim2.new(0, 180, 0, 118)
end)

-- Function to use all tools once
local function useAllTools()
    local backpack = player.Backpack
    local character = player.Character
    
    if not character then
        StatusLabel.Text = "No character found"
        return
    end
    
    local toolsUsed = 0
    
    for _, tool in pairs(backpack:GetChildren()) do
        if tool:IsA("Tool") then
            tool.Parent = character
            tool:Activate()
            task.wait(useInterval)
            tool.Parent = backpack
            toolsUsed = toolsUsed + 1
        end
    end
    
    -- Also check tools already equipped
    for _, tool in pairs(character:GetChildren()) do
        if tool:IsA("Tool") and tool.Parent == character then
            tool:Activate()
            task.wait(useInterval)
            toolsUsed = toolsUsed + 1
        end
    end
    
    StatusLabel.Text = "Used " .. toolsUsed .. " tools"
end

-- Use Once Button Click
UseOnceButton.MouseButton1Click:Connect(function()
    useAllTools()
end)

-- Auto Use Toggle
AutoUseButton.MouseButton1Click:Connect(function()
    autoUseEnabled = not autoUseEnabled
    AutoUseButton.Text = autoUseEnabled and "AUTO USE: ON" or "AUTO USE: OFF"
    AutoUseButton.BackgroundColor3 = autoUseEnabled and Color3.fromRGB(50, 200, 50) or Color3.fromRGB(80, 80, 150)
    
    if autoUseEnabled then
        StatusLabel.Text = "Auto-using tools..."
    else
        StatusLabel.Text = "Auto-use stopped"
    end
end)

-- Auto Use Loop
task.spawn(function()
    while true do
        task.wait(useInterval)
        if autoUseEnabled then
            useAllTools()
        end
    end
end)
